conf t

ip nat outside
exit

int g0/1
ip nat outside
exit

int g0/2
ip nat inside

no shut
exit

ip nat inside source static 192.168.5.2 192.168.2.2

enable

conf t
hostname HK-Edge
no cdp run
int g0/1
ip address 209.150.0.1 255.255.255.0
ip ospf message-digest-key 1 md5 password
no shutdown
exit

int g0/0
no shutdown
ip address 209.100.0.2 255.255.255.0
ip ospf message-digest-key 1 md5 password

exit
router ospf 1
network 209.150.0.0 0.0.0.255 area 0
network 209.100.0.0 0.0.0.255 area 0
passive-interface g0/0
area 0 authentication message-digest
exit
ip route 209.100.0.0 255.255.255.0 g0/0
end

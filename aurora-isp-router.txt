ISP Router:

enable

conf t
hostname router-isp

int g0/0
no shutdown
ip ospf message-digest-key 1 md5 password
ip address 209.200.0.2 255.255.255.0
exit

int g0/1
no shutdown
ip ospf message-digest-key 1 md5 password
ip address 209.150.0.2 255.255.255.0
exit

router ospf 1
network 209.200.0.0 0.0.0.255 area 0
network 209.150.0.0 0.0.0.255 area 0
area 0 authentication message-digest
end

write mem
